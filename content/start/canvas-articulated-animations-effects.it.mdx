Le animazioni articolate sono funzioni che utilizzano la funzione [`canvas.animate`](/jsdoc/pixi-vn/index/interfaces/CanvasManagerInterface#animate) per creare animazioni che possono essere applicate ai componenti canvas. Queste funzioni vengono generalmente utilizzate per creare effetti come lo scuotimento, il rimbalzo o altre animazioni complesse che coinvolgono più passaggi o componenti.

<Accordion title="Scuotimento" id="shake">

La funzione [`shakeEffect`](/jsdoc/pixi-vn/index/functions/shakeEffect) è un'animazione articolata che scuote un componente.

```ts tab="content/labels/start.label.ts"
import {
    CANVAS_APP_GAME_LAYER_ALIAS,
    newLabel,
    shakeEffect,
    showImage,
} from "@drincs/pixi-vn";

export const startLabel = newLabel("start", [
    async () => {
        await showImage("bg", "bg", { scale: 1.3 });
        await showImage("alien", "alien", { align: 0.5 });
        shakeEffect("alien"); // [!code focus]
    },
    async () => {
        shakeEffect(CANVAS_APP_GAME_LAYER_ALIAS); // [!code focus]
    },
]);
```

```ts tab="assets/index.ts"
import { AssetsManifest } from "@drincs/pixi-vn";

/**
 * Manifest for the assets used in the game.
 * You can read more about the manifest here: https://pixijs.com/8.x/guides/components/assets#loading-multiple-assets
 */
export const manifest: AssetsManifest = {
    bundles: [
        {
            name: "start",
            assets: [
                {
                    alias: "alien",
                    src: "https://pixijs.com/assets/eggHead.png",
                },
                {
                    alias: "bg",
                    src: "https://pixijs.com/assets/bg_grass.jpg",
                },
            ],
        },
    ],
};
```
