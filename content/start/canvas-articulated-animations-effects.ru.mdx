Шарнирные анимации — это функции, которые используют функцию [`canvas.animate`](/jsdoc/pixi-vn/index/interfaces/CanvasManagerInterface#animate) для создания анимаций, которые можно применять к канвас-компонентам. Эти функции обычно используются для создания таких эффектов, как тряска, подпрыгивание или другие сложные анимации, включающие несколько шагов или компонентов.

<Accordion title="Тряска" id="shake">

Функция [`shakeEffect`](/jsdoc/pixi-vn/index/functions/shakeEffect) — это шарнирная анимация, которая заставляет компонент трястись.

```ts tab="content/labels/start.label.ts"
import {
    CANVAS_APP_GAME_LAYER_ALIAS,
    newLabel,
    shakeEffect,
    showImage,
} from "@drincs/pixi-vn";

export const startLabel = newLabel("start", [
    async () => {
        await showImage("bg", "bg", { scale: 1.3 });
        await showImage("alien", "alien", { align: 0.5 });
        shakeEffect("alien"); // [!code focus]
    },
    async () => {
        shakeEffect(CANVAS_APP_GAME_LAYER_ALIAS); // [!code focus]
    },
]);
```

```ts tab="assets/index.ts"
import { AssetsManifest } from "@drincs/pixi-vn";

/**
 * Manifest for the assets used in the game.
 * You can read more about the manifest here: https://pixijs.com/8.x/guides/components/assets#loading-multiple-assets
 */
export const manifest: AssetsManifest = {
    bundles: [
        {
            name: "start",
            assets: [
                {
                    alias: "alien",
                    src: "https://pixijs.com/assets/eggHead.png",
                },
                {
                    alias: "bg",
                    src: "https://pixijs.com/assets/bg_grass.jpg",
                },
            ],
        },
    ],
};
```
