## Agregar

Para añadir un componente al canvas del juego, puedes usar [`canvas.add`](/jsdoc/pixi-vn/index/interfaces/CanvasManagerInterface#add).

```ts tab="content/labels/start.label.ts"
import { Assets, canvas, newLabel, Sprite } from "@drincs/pixi-vn";

export const startLabel = newLabel("start", [
    async () => {
        const sprite = new Sprite();
        const texture = await Assets.load("egg_head");
        sprite.texture = texture;
        canvas.add("sprite", sprite); // [!code focus]
    },
]);
```

```ts tab="assets/index.ts"
import { AssetsManifest } from "@drincs/pixi-vn";

/**
 * Manifest for the assets used in the game.
 * You can read more about the manifest here: https://pixijs.com/8.x/guides/components/assets#loading-multiple-assets
 */

    bundles: [
        {
            name: "start",
            assets: [
                {
                    alias: "egg_head",
                    src: "https://pixijs.com/assets/eggHead.png",
                },
            ],
        },
    ],
};
```

## Obtener

Para obtener un componente del canvas del juego, puedes usar [`canvas.find`](/jsdoc/pixi-vn/index/interfaces/CanvasManagerInterface#find).

```ts tab="content/labels/start.label.ts"
import { Assets, canvas, newLabel, Sprite } from "@drincs/pixi-vn";

export const startLabel = newLabel("start", [
    async () => {
        const sprite = new Sprite();
        const texture = await Assets.load("egg_head");
        sprite.texture = texture;
        canvas.add("sprite", sprite);
    },
    () => {
        const sprite = canvas.find("sprite"); // [!code focus]
        if (sprite) {
            sprite.x = 100;
            sprite.y = 100;
        }
    },
]);
```

## Eliminar todos

Para eliminar todos los componentes del canvas del juego, puedes usar [`canvas.removeAll`](/jsdoc/pixi-vn/index/interfaces/CanvasManagerInterface#removeAll).

```ts tab="content/labels/start.label.ts"
import { Assets, canvas, newLabel, Sprite } from "@drincs/pixi-vn";

export const startLabel = newLabel("start", [
    async () => {
        const texture = await Assets.load("egg_head");
        for (let i = 0; i < 3; i++) {
            const sprite = new Sprite();
            sprite.texture = texture;
            sprite.x = i * 150;
            canvas.add(`sprite${i}`, sprite);
        }
    },
    () => {
        canvas.removeAll(); // [!code focus]
    },
]);
```

<Accordion title="Añadir un listener a un evento" id="add-a-listener-to-an-event">

:::tip

Si es posible, intenta usar <DynamicLink href="/start/interface">HTML y PixiJS UI</DynamicLink> para añadir botones u otros elementos con eventos.

:::

En Pixi’VN, en comparación con PixiJS, puedes usar un [listener con el método `on`](https://pixijs.com/8.x/examples/events/click) (por ejemplo, para hacerlo clicable), pero si no usas el decorador [`@eventDecorator`](/jsdoc/pixi-vn/index/functions/eventDecorator), el evento no se registrará en el sistema de eventos de Pixi’VN, por lo que si cargas una partida guardada en la que se usó ese evento, se perderá.

```ts tab="content/canvas/events.ts"
import { eventDecorator, FederatedEvent, Sprite } from "@drincs/pixi-vn";

export default class Events {
    @eventDecorator()
    static buttonEvent(event: FederatedEvent, sprite: Sprite): void {
        switch (event.type) {
            case "pointerdown":
                sprite.scale.x *= 1.25;
                sprite.scale.y *= 1.25;
                break;
        }
    }
}
```

```ts tab="content/labels/start.label.ts"
import { newLabel, showImage } from "@drincs/pixi-vn";
import Events from "../canvas/events";

export const startLabel = newLabel("start", [
    async () => {
        const bunny = await showImage("bunny", "bunny", {
            align: 0.5,
            anchor: 0.5,
        });
        // Opt-in to interactivity
        bunny.eventMode = "static";
        // Shows hand cursor
        bunny.cursor = "pointer";
        // Pointers normalize touch and mouse (good for mobile and desktop)
        bunny.on("pointerdown", Events.buttonEvent);
    },
]);
```
